75   void  APP_LEDUpdateUSBStatus(void)
76   {
77        static  uint16_t  ledCount  =  0;
78  
79        if(USBGetSuspendState()  ==  true)
80        {
81            LED_Off(LED_USB_DEVICE_STATE);
82            return;
83        }
84  
85        switch(USBGetDeviceState())
86        {         
87            case  CONFIGURED_STATE:
88                /*  We  are  configured.   Blink  fast.
89                 *  On  for  75ms,  off  for  75ms,  then  reset/repeat.  */
90                if(ledCount  ==  1)
91                {
92                    LED_On(LED_USB_DEVICE_STATE);
93                }
94                else  if(ledCount  ==  75)
95                {
96                    LED_Off(LED_USB_DEVICE_STATE);
97                }
98                else  if(ledCount &rsaquo   150)
99                {
100                    ledCount  =  0;
101                }
102                break;
103  
104            default:
105                /*  We  aren't  configured  yet,  but  we  aren't  suspended  so  let's  blink  with
106                 *  a  slow  pulse.  On  for  50ms,  then  off  for  950ms,  then  reset/repeat.  */
107                if(ledCount  ==  1)
108                {
109                    LED_On(LED_USB_DEVICE_STATE);
110                }
111                else  if(ledCount  ==  50)
112                {
113                    LED_Off(LED_USB_DEVICE_STATE);
114                }
115                else  if(ledCount &rsaquo   950)
116                {
117                    ledCount  =  0;
118                }
119                break;
120        }
121  
122        /*  Increment  the  millisecond  counter.  */
123        ledCount++;
124   }
