27   /***  Button  Definitions  *********************************************/
28   //       S1        MCLR  reset
29   #define  S2_PORT   PORTBbits.RB4       //AN11
30   #define  S3_PORT   PORTBbits.RB5      
31  
32   #define  S2_TRIS   TRISBbits.TRISB4
33   #define  S3_TRIS   TRISBbits.TRISB5
34  
35   #define  S2_ANSEL  ANSELBbits.ANSB4
36   #define  S3_ANSEL  ANSELBbits.ANSB5
37  
38   #define  BUTTON_PRESSED       0
39   #define  BUTTON_NOT_PRESSED   1
40  
41   #define  PIN_INPUT            1
42   #define  PIN_OUTPUT           0
43  
44   #define  PIN_DIGITAL          0
45   #define  PIN_ANALOG           1
46  
47   /*********************************************************************
48   *  Function:  bool  BUTTON_IsPressed(BUTTON  button);
49   *
50   *  Overview:  Returns  the  current  state  of  the  requested  button
51   *
52   *  PreCondition:  button  configured  via  BUTTON_SetConfiguration()
53   *
54   *  Input:  BUTTON  button  -  enumeration  of  the  buttons  available  in
55   *         this  demo.   They  should  be  meaningful  names  and  not  the  names 
56   *         of  the  buttons  on  the  silkscreen  on  the  board  (as  the  demo 
57   *         code  may  be  ported  to  other  boards).
58   *          i.e.  -  ButtonIsPressed(BUTTON_SEND_MESSAGE);
59   *
60   *  Output:  TRUE  if  pressed;  FALSE  if  not  pressed.
61   *
62   ********************************************************************/
63   bool  BUTTON_IsPressed(BUTTON  button)
64   {
65        switch(button)
66        {
67            case  BUTTON_S2:
68                return  (  (S2_PORT  ==  BUTTON_PRESSED)  ?  true  :  false);
69                          
70            case  BUTTON_S3:
71                return  (  (S3_PORT  ==  BUTTON_PRESSED)  ?  true  :  false);
72  
73            case  BUTTON_NONE:
74                return  false;
75        }
76       
77        return  false;
78   }
79  
80   /*********************************************************************
81   *  Function:  void  BUTTON_Enable(BUTTON  button);
82   *
83   *  Overview:  Returns  the  current  state  of  the  requested  button
84   *
85   *  PreCondition:  button  configured  via  BUTTON_SetConfiguration()
86   *
87   *  Input:  BUTTON  button  -  enumeration  of  the  buttons  available  in
88   *         this  demo.   They  should  be  meaningful  names  and  not  the  names
89   *         of  the  buttons  on  the  silkscreen  on  the  board  (as  the  demo
90   *         code  may  be  ported  to  other  boards).
91   *          i.e.  -  ButtonIsPressed(BUTTON_SEND_MESSAGE);
92   *
93   *  Output:  None
94   *
95   ********************************************************************/
96   void  BUTTON_Enable(BUTTON  button)
97   {
98        switch(button)
99        {
100            case  BUTTON_S2:
101                S2_TRIS  =  PIN_INPUT;
102                S2_ANSEL  =  PIN_DIGITAL;
103                break;
104                          
105            case  BUTTON_S3:
106                S3_TRIS  =  PIN_INPUT;
107                S3_ANSEL  =  PIN_DIGITAL;
108                break;
109  
110            case  BUTTON_NONE:
111                break;
112        }
113   }
